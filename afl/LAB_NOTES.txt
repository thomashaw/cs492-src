Lab #2
------
  resize window to have at least 80x25 dimensions
  wget http://thefengs.com/wuchang/courses/cs492/fuzz/program_deadbeef.c
  mkdir in out
  afl-clang-fast -o program_deadbeef program_deadbeef.c
  echo 'h' > in/foo

  # SINGLE CPU
  #   Let run for 30 minutes on a micro instance
  #   Perform a hexdump on crashing input (out/crashes/id*)
  afl-fuzz -m 2G -i in -o out -- ./program_deadbeef

  # 4 CPUs in parallel
  #   Let run for 5 minutes on a 4 vCPU
  #   Perform a hexdump on crashing input (sync_dir/*/crashes/id*)
  mkdir sync_dir
  tmux
  # Then do... <Control-b>+c three times to create three more windows
  # To switch between windows <Control-b> n or <Control-b> p
  afl-fuzz -m 2G -i in -o sync_dir -M fuzzer01 -- ./program_deadbeef
  <Control-b> n
  afl-fuzz -m 2G -i in -o sync_dir -S fuzzer02 -- ./program_deadbeef
  <Control-b> n
  afl-fuzz -m 2G -i in -o sync_dir -S fuzzer03 -- ./program_deadbeef
  <Control-b> n
  afl-fuzz -m 2G -i in -o sync_dir -S fuzzer04 -- ./program_deadbeef
